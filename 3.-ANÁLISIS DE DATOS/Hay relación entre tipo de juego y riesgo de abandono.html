SELECT 
    favorite_game,
    COUNT(*) as total_usuarios,
    SUM(churn_risk) as usuarios_en_riesgo,
    ROUND(AVG(churn_risk) * 100, 2) as porcentaje_riesgo,
    ROUND(AVG(deposit_count), 1) as promedio_depositos,
    ROUND(AVG(avg_deposit), 2) as ticket_promedio,
    ROUND(AVG(total_game_time), 1) as horas_promedio,
    ROUND(AVG(total_sessions), 1) as sesiones_promedio
FROM casino_users
WHERE deposit_count > 0
GROUP BY favorite_game
HAVING COUNT(*) >= 2
ORDER BY porcentaje_riesgo DESC, total_usuarios DESC
