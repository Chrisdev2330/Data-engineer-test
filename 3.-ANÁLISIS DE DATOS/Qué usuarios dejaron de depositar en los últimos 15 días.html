SELECT 
    user_id,
    country,
    CAST(julianday('now') - julianday(last_deposit_date) AS INTEGER) as dias_sin_depositar,
    deposit_count,
    ROUND(avg_deposit, 2) as ticket_promedio,
    last_deposit_date,
    total_sessions,
    ROUND(total_game_time, 2) as horas_totales,
    favorite_game,
    device_type,
    CASE 
        WHEN avg_deposit > 100 AND deposit_count > 10 THEN 'Prioridad Alta'
        WHEN avg_deposit > 50 AND deposit_count > 5 THEN 'Prioridad Media'
        ELSE 'Prioridad Baja'
    END as prioridad_reactivacion,
    churn_risk
FROM casino_users
WHERE last_deposit_date IS NOT NULL
  AND julianday('now') - julianday(last_deposit_date) > 15
  AND deposit_count > 0
ORDER BY 
    CASE 
        WHEN avg_deposit > 100 AND deposit_count > 10 THEN 1
        WHEN avg_deposit > 50 AND deposit_count > 5 THEN 2
        ELSE 3
    END,
    avg_deposit DESC,
    dias_sin_depositar ASC
