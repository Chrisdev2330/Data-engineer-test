SELECT 
    user_id,
    country,
    deposit_count,
    ROUND(
        total_game_time / 
        CASE 
            WHEN (julianday('now') - julianday(registration_date)) < 30 THEN 1
            ELSE (julianday('now') - julianday(registration_date)) / 30 
        END, 
        2
    ) as horas_mensuales,
    ROUND(avg_deposit, 2) as ticket_promedio,
    total_sessions,
    favorite_game,
    device_type,
    CASE 
        WHEN deposit_count > 20 THEN 'VIP'
        WHEN deposit_count > 10 THEN 'Premium'
        WHEN deposit_count > 5 THEN 'Regular Plus'
        ELSE 'Regular'
    END as segmento
FROM casino_users
WHERE deposit_count > 3
  AND total_game_time / 
      CASE 
          WHEN (julianday('now') - julianday(registration_date)) < 30 THEN 1
          ELSE (julianday('now') - julianday(registration_date)) / 30 
      END > 5
ORDER BY deposit_count DESC, horas_mensuales DESC
